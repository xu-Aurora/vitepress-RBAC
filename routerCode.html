动态路由表 ​

js
const dynamicRouter = [
  {
    path: '/onePage',
    name: 'OnePage',
    meta: { title: '一级页面' },
    component: 'onePage'
  },
  {
    path: '/menu',
    meta: {
      title: '菜单',
    },
    children: [
      {
        path: '/menu/menu1',
        name: 'Menu1',
        // menu 和 menu1 都表示文件夹, 所有路由文件默认取名 index.vue
        component: 'menu/menu1',
        meta: {
          title: '菜单一',
        }
      },
      {
        path: '/menu/menu2',
        name: 'Menu2',
        component: 'menu/menu2',
        meta: {
          title: '菜单二',
        }
      }
    ]
  }
]

TIP

这些路由都是由后端返回。

动态路由组装 ​

js
import router from '@/routers/index';

// 引入 views 文件下所有 vue 文件
const modules = import.meta.glob('@/views/**/*.vue')

// 通过 router.addRoute 递归生成路由
function generator(routes: any[]) {
  routes.forEach((route) => {
    if (route.children) {
      generator(route.children)
    }
    if (route.component && typeof route.component === 'string') {
      route.component = modules['/src/views/' + route.component + '/index.vue']
    }
    // 在 layout 路由下添加子路由
    router.addRoute("layout", route)
  })
}

TIP

为了组装路由的 component 属性匹配 views 的路由文件。
